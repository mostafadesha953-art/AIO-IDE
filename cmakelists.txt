cmake_minimum_required(VERSION 3.10)
project(AIO_Smart_IDE LANGUAGES CXX) # تأكيد اللغة C++

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")

# جلب المكتبات
find_package(CURL REQUIRED)
find_package(nlohmann_json REQUIRED)

# البحث عن الملفات وإجبار النظام على اعتبارها C++
file(GLOB_RECURSE SOURCES ".cpp" ".h")

# --- السطر السحري لحل المشكلة ---
set_target_properties(AIO_IDE PROPERTIES LINKER_LANGUAGE CXX)

add_executable(AIO_IDE ${SOURCES})

target_include_directories(AIO_IDE PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}")
target_link_libraries(AIO_IDE PRIVATE CURL::libcurl nlohmann_json::nlohmann_json d3d11 dwmapi imm32 winmm)
